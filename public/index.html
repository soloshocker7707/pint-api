INTEGRATE IN SECONDS.

RenderKit is designed to be a drop-in replacement for any headless browser infrastructure. Simple, stateless, and high-performance.

POST /v1/screenshot/capture 
curl -X POST https://api.renderkit.com/v1/screenshot/capture \
-H "x-api-key: YOUR_API_KEY" \
-H "Content-Type: application/json" \
-d '{
  "url": "https://google.com",
  "wait": "smart",
  "clean": true,
  "freezeAnimations": true
}'
renderkit.js 
const axios = require('axios');

const render = async () => {
  const res = await axios.post('https://api.renderkit.com/v1/screenshot/capture', {
    url: 'https://example.com',
    wait: 'smart'
  }, {
    headers: { 'x-api-key': 'YOUR_API_KEY' }
  });
  
  console.log(res.data.image_base64);
};
capture.py 
import requests

payload = {
    "url": "https://example.com",
    "wait": "smart"
}

headers = { "x-api-key": "YOUR_API_KEY" }

response = requests.post(
    "https://api.renderkit.com/v1/screenshot/capture", 
    json=payload, 
    headers=headers
)

print(response.json())
